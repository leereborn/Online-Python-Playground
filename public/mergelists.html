Merge Lists

Write an immutable function that merges the following inputs into a single list. 


def merge(l_origin, l_add, l_del):
    res_set = set(l_origin)
    res_set.update(l_add)
    res_set.difference_update(l_del)

    res_list = list(res_set)
    res_list.sort(reverse=True)
    res_list.sort(key = lambda x: len(x), reverse=True)

    return res_list

#Feel free to modify the input lists below
original_list = ['one', 'two', 'three']
add_list = ['one', 'two', 'five', 'six']
delete_list = ['two', 'five']
print(merge(original_list, add_list, delete_list))
